1function [hr,rr] = extract_vitals_tk1(frames_head, frames_body, timestamps, fps, block_size)
  2%#codegen
  3% assert(isa(frames,'double'))
  4% assert(isa(block_size,'double'))
  5
  6% fprintf('%d,%d',size(frames_head, 1), size(frames_head, 2));
  7% fprintf('%d,%d',size(frames_body, 1), size(frames_body, 2));

  9
 10if nargin < 2
 11  block_size = 4;
 12end
 13
 14c_interp = 1000;
 15fps_target = 30;
 16timestamps = floor(timestamps * c_interp) / c_interp; % ???
 17
 18Bbody = permute(frames_body, [2 3 1]);  % make it h x w x t
 19Bbody = imresize(Bbody, 1/block_size, 'box');
 20% Rbody = reshape(permute(Bbody, [3 1 2]),size(Bbody,3),[]);
 21assert(size(Bbody,3) == length(timestamps))
 22Rbody = reshape(permute(Bbody, [3 1 2]), numel(timestamps), []);
 23% Interpolate between uneven timestamped measurements
 24Rbody_interp = zeros(numel(min(timestamps):1/fps_target:max(timestamps)), size(Rbody,2));
 25for i  = 1 : size(Rbody,2)
 26  Rbody_interp(:,i) = interp1(timestamps, Rbody(:,i)', min(timestamps):1/fps_target:max(timestamps));
 27end
 28Abody = -log(1+Rbody_interp);
 29
 30Bhead = permute(frames_head, [2 3 1]);  % make it h x w x t
 31Bhead = imresize(Bhead, 1/block_size, 'box');
 32assert(size(Bhead,3) == length(timestamps))
 33Rhead = reshape(permute(Bhead, [3 1 2]), numel(timestamps), []);
 34% Interpolate between uneven timestamped measurements
 35Rhead_interp = zeros(numel(min(timestamps):1/fps_target:max(timestamps)), size(Rhead,2));
 36for i  = 1 : size(Rhead,2)
 37  Rhead_interp(:,i) = interp1(timestamps, Rhead(:,i)', min(timestamps):1/fps_target:max(timestamps));
 38end
 39Ahead = -log(1+Rhead_interp);
 40
 41[T,~] = size(Ahead);
 42
 43%%
 44% breathing = mean(Abody,2);
 45% heartrate = mean(Ahead,2);
 46
 47% Get rid of breathing component of heartrate signal.
 48low_freq = 50/60; % low_freq = 30/60;
 49high_freq = 80/60; % high_freq = 350/60;
 50% x_ts = timeseries(Ahead, [0:T-1]./60);
 51% x_filt = idealfilter(x_ts, [low_freq high_freq], 'pass');

 53
 54% butter() requires constant valued inputs, so put in if statements.
 55if abs(fps-30) < 0.1
 56  nyq = 15;
 57  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 58elseif abs(fps-60) < 0.1
 59  nyq = 30;
 60  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 61else
 62  error('Unsupported frame rate needed for constant nyq criterion in butter(). Please update if statements in the code.');
 63end

 65for i = 1:size(Ahead_filt,2)
 66  Ahead_filt(:,i) = filter(b, a, Ahead(:,i));
 67end
 68
 69low_freq = 5/60;
 70high_freq = 25/60;
 71if abs(fps-30) < 0.1
 72  nyq = 15;  % fs/2
 73  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 74elseif abs(fps-60) < 0.1
 75  nyq = 30;  % fs/2
 76  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 77else
 78  error('Unsupported frame rate needed for constant nyq criterion in butter(). Please update if statements in the code.');
 79end
 80% Abody = zeros(size(Abody));
 81% for i = 1:size(Abody,2)
 82%   Abody(:,i) = filter(b, a, Abody(:,i));
 83% end
 84
 85
 86
 87
 88% Do weighted average of signals based on (inverse) entropy to yield a
 89% signal (heartrate and breathing).
 90
 91% power = (a^2+b^2)/N
 92% Should be same as 1/N*abs(Y)^2
 93npts = size(Abody,1);
 94Fheartrate = zeros(floor(npts/2), size(Ahead,2));
 95Fbreathing = zeros(floor(npts/2), size(Abody,2));
 96
 97Y = fft(Abody, npts);
 98Pyy = Y.*conj(Y)/npts;
 99freq = fps/npts * (0:floor(npts/2)-1);
100Fbreathing = Pyy(1:floor(npts/2),:);
101Y = fft(Ahead, npts);
102Pyy = Y.*conj(Y)/npts;
103freq = fps/npts * (0:floor(npts/2)-1);
104Fheartrate = Pyy(1:floor(npts/2),:);
105
106% [freq, Fbreathing] = plot_power_spectrum(Abody, 60);
107% [freq, Fheartrate] = plot_power_spectrum(Ahead_filt, 60);
108idx1 = find(freq>5/60, 1, 'first') - 1;
109idx2 = find(freq>25/60, 1, 'first');
110Fbreathing([1:idx1, idx2:end],:) = 0;
111Fheartrate(1,:) = 0;
112% Fbreathing(1,:) = zeros(1, size(Fbreathing,2));
113% Fheartrate(1,:) = zeros(1, size(Fheartrate,2));
114
115HRentr = get_spectral_entropy(Fheartrate);
116whr = 1-HRentr;
117whr = whr./sum(whr);
118heartrate = Ahead_filt * whr';
119% trust neighborhoods of high weight
120RRentr = get_spectral_entropy(Fbreathing);
121wrr = 1-RRentr;
122wrr = wrr./sum(wrr);
123B = 1-reshape(wrr,size(Bbody(:,:,1)));
124% B2 = nlfilter(B,[2 2],'sliding',@min);
125% B(1:end-1,1:end-1) = B2(1:end-1,1:end-1);
126for i = 2:2:size(B,1)-1
127  for j = 2:2:size(B,2)-1
128    patch = B(i-1:i+1, j-1:j+1);
129    B(i,j) = min(patch(:));
130  end
131end
132wrr = B(:);
133
134
135% Use fft to get rid of phase differences between signals.
136F = abs(fft(Abody));
137F(1) = 0;
138F([1:idx1, idx2:end],:) = 0;
139% breathing = ifft(F * wrr');
140breathing = F * wrr(:);
141% breathing = Abody * wrr';
142
143sigmam = 1.4e-3;
144sigmap = 2.3e-5;
145R = exp(-heartrate)-1;
146xhat = kalmanfilt(R', sigmap, sigmam);
147heartrate = -log(xhat+1);

149[~,maxidx] = max(whr);
150heartrate_best = Ahead_filt(:,maxidx);
151R = exp(-heartrate_best)-1;
152xhat = kalmanfilt(R', sigmap, sigmam);
153heartrate_best = -log(xhat+1);

162% imshow(flipud(reshape(wrr,[H W])),[]);
163% title('Breathing weights')
