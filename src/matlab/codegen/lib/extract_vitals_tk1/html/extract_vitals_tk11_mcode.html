1function [hr,rr] = extract_vitals_tk1(frames_head, frames_body, fs, block_size)
  2%#codegen
  3% assert(isa(frames,'double'))
  4% assert(isa(block_size,'double'))
  5
  6if nargin < 2
  7  block_size = 6;
  8end
  9
 10Bbody = imresize(frames_body,1/block_size,'box');
 11Rbody = reshape(permute(Bbody,[3 1 2]),size(Bbody,3),[]);
 12Abody = -log(1+Rbody);
 13
 14Bhead = imresize(frames_head,1/block_size,'box');
 15Rhead = reshape(permute(Bhead,[3 1 2]),size(Bhead,3),[]);
 16Ahead = -log(1+Rhead);
 17
 18[T,~] = size(Ahead);
 19
 20%%
 21% breathing = mean(Abody,2);
 22% heartrate = mean(Ahead,2);
 23
 24% Get rid of breathing component of heartrate signal.
 25low_freq = 30/60;
 26high_freq = 350/60;
 27% x_ts = timeseries(Ahead, [0:T-1]./60);
 28% x_filt = idealfilter(x_ts, [low_freq high_freq], 'pass');
 29% Ahead_filt = x_filt.Data;
 30
 31% butter() requires constant valued inputs, so put in if statements.
 32if abs(fs-30) < 0.1
 33  nyq = 15;
 34  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 35elseif abs(fs-60) < 0.1
 36  nyq = 30;
 37  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 38else
 39  error('Unsupported frame rate needed for constant nyq criterion in butter(). Please update if statements in the code.');
 40end
 41
 42Ahead_filt = zeros(size(Ahead));
 43for i = 1:size(Ahead_filt,2)
 44  Ahead_filt(:,i) = filter(b, a, Ahead(:,i));
 45end
 46
 47
 48
 49
 50% Do weighted average of signals based on (inverse) entropy to yield a
 51% signal (heartrate and breathing).
 52
 53% power = (a^2+b^2)/N
 54% Should be same as 1/N*abs(Y)^2
 55npts = size(Abody,1);
 56Fheartrate = zeros(floor(npts/2), size(Ahead,2));
 57Fbreathing = zeros(floor(npts/2), size(Abody,2));
 58
 59Y = fft(Abody, npts);
 60Pyy = Y.*conj(Y)/npts;
 61freq = fs/npts * (0:floor(npts/2)-1);
 62Fbreathing = Pyy(1:floor(npts/2),:);
 63Y = fft(Ahead, npts);
 64Pyy = Y.*conj(Y)/npts;
 65freq = fs/npts * (0:floor(npts/2)-1);
 66Fheartrate = Pyy(1:floor(npts/2),:);
 67
 68% [freq, Fbreathing] = plot_power_spectrum(Abody, 60);
 69% [freq, Fheartrate] = plot_power_spectrum(Ahead_filt, 60);
 70Fbreathing(1,:) = 0;
 71Fheartrate(1,:) = 0;
 72% Fbreathing(1,:) = zeros(1, size(Fbreathing,2));
 73% Fheartrate(1,:) = zeros(1, size(Fheartrate,2));
 74
 75HRentr = get_spectral_entropy(Fheartrate);
 76whr = 1-HRentr;
 77whr = whr./sum(whr);
 78heartrate = Ahead_filt * whr';
 79
 80RRentr = get_spectral_entropy(Fbreathing);
 81wrr = 1-RRentr;
 82wrr = wrr./sum(wrr);
 83% Use fft to get rid of phase differences between signals.
 84F = abs(fft(Abody));
 85F(1,:) = 0;
 86breathing = ifft(F * wrr');
 87
 88sigmam = 1.4e-3;
 89sigmap = 2.3e-5;
 90R = exp(-heartrate)-1;
 91xhat = kalmanfilt(R', sigmap, sigmam);
 92heartrate = -log(xhat+1);
 93
 94[~,maxidx] = max(whr);
 95heartrate_best = Ahead_filt(:,maxidx);
 96R = exp(-heartrate_best)-1;
 97xhat = kalmanfilt(R', sigmap, sigmam);
 98heartrate_best = -log(xhat+1);
 99% 
100% figure;
101% subplot(1,2,1),
102% [H,W,~] = size(Bhead);
103% imshow(flipud((reshape(whr,[H W]))),[])
104% title('Pulse weights')
105% subplot(1,2,2)
106% [H,W,~] = size(Bbody);
107% imshow(flipud(reshape(wrr,[H W])),[]);
108% title('Breathing weights')
109
110%% Find vitals
111[freq, Fbreathing] = plot_power_spectrum(breathing, 60);
112[freq, Fheartrate] = plot_power_spectrum(heartrate, 60);
113
114Fbreathing(1,:) = 0;
115Fheartrate(1,:) = 0;
116
117[~,maxidx] = max(Fheartrate);
118hr = freq(maxidx);
119[~,maxidx] = max(Fbreathing);
120rr = freq(maxidx);
121
122% t = [0:numel(heartrate_best)-1] ./ 60;
123% figure;
124% subplot(2,2,1), plot(t,heartrate_best), title('Heart rate')
125% hold on, plot(t,heartrate,'--k')
126% xlabel('time (s)')
127% ylabel('blood volume (a.u.)')
128% legend('strongest signal','weighted average')
129% 
130% subplot(2,2,2), plot(t,breathing), title('Breathing')
131% xlabel('time (s)')
132% ylabel('breathing amplitude (a.u.)')
133% 
134% subplot(2,2,3), plot(60*freq,Fheartrate), title(sprintf('HR=%u bpm',round(hr*60)))
135% [~,maxidx] = max(Fheartrate);
136% hold on, plot(60*freq(maxidx),Fheartrate(maxidx),'or')
137% xlim([0 300])
138% xlabel('frequency (1/min)')
139% ylabel('spectral power (dB)')
140% 
141% subplot(2,2,4), plot(60*freq,Fbreathing), title(sprintf('RR=%u bpm',round(rr*60)))
142% [~,maxidx] = max(Fbreathing);
143% hold on, plot(60*freq(maxidx),Fbreathing(maxidx),'or')
144% xlim([0 300])
145% xlabel('frequency (1/min)')
146% ylabel('spectral power (dB)')
