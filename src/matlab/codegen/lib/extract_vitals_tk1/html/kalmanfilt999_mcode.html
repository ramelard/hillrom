1% KALMANFILT   Kalman filter
 2%   xest = kalmanfilt(xhat, sigmap, sigmam)
 3%     For now, assumes 2D position, velocity setup.
 4% For codegen: 
 5%   codegen -args {coder.typeof(double(0), [1 Inf]) double(0) double(0)} kalmanfilt.m
 6%#codegen
 7function xest = kalmanfilt(z, sigmap, sigmam)
 8
 9if nargin < 2
10  sigmap = 3;  % process error, stddev
11  sigmam = 0.01;  % measurement error, stddev
12end
13if nargin < 1
14  t = 0:.001:1;
15  x = sin(2*pi*3*t);  % true signal (cannot observe)
16  xhat = x+sigmap*randn(size(x));  % actual signal output
17  z = xhat + sigmam*randn(size(xhat));  % measured signal
18end
19
20Q = [1/4 1/2; 1/2 1]*sigmap^2;  % process noise covariance
21R = sigmam^2;  % measurement noise variance
22
23A = [1 1; 0 1];  % state transition matrix
24B = [1/2; 1];
25u = 0;
26H = [1 0];  % observation matrix
27
28xprior = [z(1); 0];  % x,v initially
29Pprior = [1 0; 0 1];  % we are certain, thus 0 stddev
30
31% Posterior estimate of signal x
32xpost = zeros(2,size(z,2));
33Ppost = [];
34for k = 1:size(z,2)
35  % Correct/Update (supports multiple observations)
36  for o = 1:size(z,1)  % number of observations
37    K = Pprior*H'*pinv(H*Pprior*H' + R);  % Kalman gain
38    xpost(:,k) = real(xprior + K*(z(o,k)-H*xprior));  % estimate x using msmt data
39    Ppost = (eye(2) - K*H)*Pprior;  % update posterior error covariance
40  end
41
42  % Predict
43  xprior = A*xpost(:,k) + B*u;  % build prior based on previous position
44  Pprior = A*Ppost*A' + Q;  % build prior error covariance
45end
46
47
48% figure, hold on
49% plot(x,'--k')
50% plot(xhat,'-r')
51% plot(z,'-g')
52% plot(xpost(1,:),'b','LineWidth',2)
53% legend('x','xhat','z','xest')
54
55xest = xpost(1,:);
