1%PLOT_POWER_SPECTRUM Plot frequency power spectrum
 2%
 3% [freq,fpower] = plot_power_spectrum(y, sampling_rate)
 4%
 5% From http://www.mathworks.com/products/matlab/examples.html?file=/products/demos/shipping/matlab/fftdemo.html
 6% y - signal (columns are observations)
 7function [freq,power] = plot_power_spectrum(y, sampling_rate, ax, linecolor)
 8
 9if nargin < 4
10  linecolor = 'b';
11end
12
13if size(y,1) == 1
14  sigs = y(:);
15else
16  sigs = y;
17end
18
19% if size(y,2) > size(y,1)
20%   warning(sprintf('Are you sure columns are observations? (r%u,c%u)', size(y,1), size(y,2)))
21% end
22
23% npts = numel(y);
24npts = size(sigs,1);
25
26Y = fft(sigs, npts);
27% power = (a^2+b^2)/N
28% Should be same as 1/N*abs(Y)^2
29Pyy = Y.*conj(Y)/npts;
30
31f = sampling_rate/npts * (0:floor(npts/2)-1);
32% nth bin is n*Fs/N, +1 for index offset
33f30bpm = floor(30/60*npts/sampling_rate) + 1;
34f200bpm = floor(200/60*npts/sampling_rate) + 1;
35
36if nargout == 0 || nargin == 3
37  if nargin < 3
38    figure;
39    ax = gca;
40  end
41  plot(ax, f, Pyy(1:floor(npts/2)), 'Color', linecolor)
42  % xlim([f(f30bpm) f(f200bpm)])
43  xlim(ax, [30/60 200/60])
44  ylim(ax, [0 max(Pyy(f30bpm : f200bpm))])
45  title('Power spectral density')
46  xlabel('Frequency (Hz)')
47end
48
49if nargout > 0
50  freq = f;
51  power = Pyy(1:floor(npts/2),:);
52end
