  538             /*  Core - second dimension */
  539             linearInds = weights->size[0] * outCInd;
  540             for (k = 0; k < weights->size[0]; k++) {
  541               c = (indices->data[linearInds] - 1) * in_->size[0];
  542               sumVal1 += weights->data[linearInds] * in_->data[idx + c];
  543               linearInds++;
  544             }
  545   
  546             out->data[(inRInd + out->size[0] * outCInd) + out->size[0] * out->
  547               size[1] * pInd] = sumVal1;
  548           }
  549         }
  550       }
  551     }
  552   }

  554   /*
  555    * Arguments    : const emxArray_real_T *Ain
  556    *                double varargin_1
  557    *                emxArray_real_T *Bout
  558    * Return Type  : void
  559    */
  560   void imresize(const emxArray_real_T *Ain, double varargin_1, emxArray_real_T
  561                 *Bout)
  562   {
  563     double scale[2];
  564     int k;
  565     double scaleOrSize[2];
  566     double outputSize[2];
  567     int iidx[2];
  568     emxArray_real_T *APartialResize;
  569     emxArray_real_T *weights;
  570     emxArray_int32_T *indices;
  571     scale[0] = varargin_1;
  572     scale[1] = varargin_1;
  573     for (k = 0; k < 2; k++) {
  574       outputSize[k] = (double)(unsigned int)Ain->size[k] * scale[k];
  575     }
  576   
  577     for (k = 0; k < 2; k++) {
  578       scaleOrSize[k] = scale[k];
  579       outputSize[k] = ceil(outputSize[k]);
  580     }
  581   
  582     sort(scaleOrSize, iidx);
  583     for (k = 0; k < 2; k++) {
  584       scaleOrSize[k] = iidx[k];
  585     }
  586   
  587     emxInit_real_T(&APartialResize, 3);
  588     emxInit_real_T1(&weights, 2);
  589     emxInit_int32_T1(&indices, 2);
  590   
  591     /*  Resize first dimension */
  592     contributions(Ain->size[(int)scaleOrSize[0] - 1], outputSize[(int)scaleOrSize
  593                   [0] - 1], scale[(int)scaleOrSize[0] - 1], weights, indices);
  594     resizeAlongDim(Ain, scaleOrSize[0], weights, indices, APartialResize);
  595   
  596     /*  Resize second dimension */
  597     contributions(Ain->size[(int)scaleOrSize[1] - 1], outputSize[(int)scaleOrSize
  598                   [1] - 1], scale[(int)scaleOrSize[1] - 1], weights, indices);
  599     resizeAlongDim(APartialResize, scaleOrSize[1], weights, indices, Bout);
  600     emxFree_int32_T(&indices);
  601     emxFree_real_T(&weights);
  602     emxFree_real_T(&APartialResize);
  603   }
