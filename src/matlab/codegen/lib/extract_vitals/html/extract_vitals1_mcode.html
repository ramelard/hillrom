1function [hr,rr] = extract_vitals(frames, face_rect, block_size)
  2%#codegen
  3% assert(isa(frames,'double'))
  4% assert(isa(block_size,'double'))

  6if nargin < 2
  7  block_size = 6;

  9

 13
 14% fprintf(1,'Finding face...')
 15% % Need to make face pointing up so it works with Viola-Jones.
 16% frame = flipud(frames(:,:,1));
 17% % Viola-Jones cascade face detector. Min size chosen empirically based on
 18% % optical setup.
 19% faceDetector = vision.CascadeObjectDetector('MinSize',[75 75]);
 20% % faceDetector = vision.CascadeObjectDetector('ProfileFace','MinSize',[75 75]);
 21% face_rect = step(faceDetector, frame);
 22% 
 23% if isempty(face_rect)
 24%   warning('No face detected. Using coarse ROIs.')
 25%   body = frames(2:304,2:678,:);
 26%   head = frames(443:851, 110:521, :);
 27% else
 28  [H,W,T] = size(frames);
 29  x = face_rect(1);
 30  y = H-face_rect(2);
 31  w = face_rect(3);
 32  h = face_rect(4);
 33
 34  % Draw the returned bounding box around the detected face.
 35%   videoOut = insertObjectAnnotation(frame,'rectangle',face_rect,'Face');

 37

 40  head = frames(y-h : y, x:x+w, :);
 41  % The rest is body.
 42  body = frames(1 : y-h*2, 1 : end, :);
 43% end
 44  
 45Bbody = imresize(body,1/block_size,'box');
 46Rbody = reshape(permute(Bbody,[3 1 2]),size(Bbody,3),[]);
 47Abody = -log(1+Rbody);
 48
 49Bhead = imresize(head,1/block_size,'box');
 50Rhead = reshape(permute(Bhead,[3 1 2]),size(Bhead,3),[]);
 51Ahead = -log(1+Rhead);
 52
 53[T,~] = size(Ahead);
 54
 55%%
 56% breathing = mean(Abody,2);
 57% heartrate = mean(Ahead,2);
 58
 59% Get rid of breathing component of heartrate signal.
 60low_freq = 30/60;
 61high_freq = 350/60;
 62% x_ts = timeseries(Ahead, [0:T-1]./60);
 63% x_filt = idealfilter(x_ts, [low_freq high_freq], 'pass');
 64% Ahead_filt = x_filt.Data;
 65
 66fs = 60;
 67nyq = fs/2;
 68[b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 69Ahead_filt = zeros(size(Ahead));
 70for i = 1:size(Ahead_filt,2)
 71  Ahead_filt(:,i) = filter(b, a, Ahead(:,i));
 72end
 73
 74
 75
 76
 77% Do weighted average of signals based on (inverse) entropy to yield a
 78% signal (heartrate and breathing).
 79
 80% power = (a^2+b^2)/N
 81% Should be same as 1/N*abs(Y)^2
 82npts = size(Abody,1);
 83Fheartrate = zeros(floor(npts/2), size(Ahead,2));
 84Fbreathing = zeros(floor(npts/2), size(Abody,2));
 85
 86Y = fft(Abody, npts);
 87Pyy = Y.*conj(Y)/npts;
 88freq = 60/npts * (0:floor(npts/2)-1);
 89Fbreathing = Pyy(1:floor(npts/2),:);
 90Y = fft(Ahead, npts);
 91Pyy = Y.*conj(Y)/npts;
 92freq = 60/npts * (0:floor(npts/2)-1);
 93Fheartrate = Pyy(1:floor(npts/2),:);
 94
 95% [freq, Fbreathing] = plot_power_spectrum(Abody, 60);
 96% [freq, Fheartrate] = plot_power_spectrum(Ahead_filt, 60);
 97Fbreathing(1,:) = 0;
 98Fheartrate(1,:) = 0;
 99% Fbreathing(1,:) = zeros(1, size(Fbreathing,2));
100% Fheartrate(1,:) = zeros(1, size(Fheartrate,2));
101
102HRentr = get_spectral_entropy(Fheartrate);
103whr = 1-HRentr;
104whr = whr./sum(whr);
105heartrate = Ahead_filt * whr';
106
107RRentr = get_spectral_entropy(Fbreathing);
108wrr = 1-RRentr;
109wrr = wrr./sum(wrr);
110% Use fft to get rid of phase differences between signals.
111F = abs(fft(Abody));
112F(1,:) = 0;
113breathing = ifft(F * wrr');
114
115sigmam = 1.4e-3;
116sigmap = 2.3e-5;
117R = exp(-heartrate)-1;
118xhat = kalmanfilt(R', sigmap, sigmam);
119heartrate = -log(xhat+1);
120
121[~,maxidx] = max(whr);
122heartrate_best = Ahead_filt(:,maxidx);
123R = exp(-heartrate_best)-1;
124xhat = kalmanfilt(R', sigmap, sigmam);
125heartrate_best = -log(xhat+1);
126% 
127% figure;
128% subplot(1,2,1),
129% [H,W,~] = size(Bhead);
130% imshow(flipud((reshape(whr,[H W]))),[])
131% title('Pulse weights')
132% subplot(1,2,2)
133% [H,W,~] = size(Bbody);
134% imshow(flipud(reshape(wrr,[H W])),[]);
135% title('Breathing weights')
136
137%% Find vitals
138[freq, Fbreathing] = plot_power_spectrum(breathing, 60);
139[freq, Fheartrate] = plot_power_spectrum(heartrate, 60);
140
141Fbreathing(1,:) = 0;
142Fheartrate(1,:) = 0;
143
144[~,maxidx] = max(Fheartrate);
145hr = freq(maxidx);
146[~,maxidx] = max(Fbreathing);
147rr = freq(maxidx);

152% hold on, plot(t,heartrate,'--k')
153% xlabel('time (s)')
154% ylabel('blood volume (a.u.)')
155% legend('strongest signal','weighted average')
