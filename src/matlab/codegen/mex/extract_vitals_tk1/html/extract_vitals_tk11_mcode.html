1function [hr,rr] = extract_vitals_tk1(frames_head, frames_body, timestamps, fps, block_size)

 10if nargin < 2

 13
 14c_interp = 1000;

 18% Bbody = permute(frames_body, [2 3 1]);  % make it h x w x t

 21assert(size(Bbody,3) == length(timestamps))
 22Rbody = reshape(permute(Bbody, [3 1 2]), numel(timestamps), []);
 23% Interpolate between uneven timestamped measurements
 24Rbody_interp = zeros(numel(min(timestamps):1/fps_target:max(timestamps)), size(Rbody,2));
 25for i  = 1 : size(Rbody,2)
 26  Rbody_interp(:,i) = interp1(timestamps, Rbody(:,i)', min(timestamps):1/fps_target:max(timestamps));

 38  Rhead_interp(:,i) = interp1(timestamps, Rhead(:,i)', min(timestamps):1/fps_target:max(timestamps));
 39end

 60  nyq = 30;

 76  nyq = 30;  % fs/2
 77  [b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 78else
 79  error('Unsupported frame rate needed for constant nyq criterion in butter(). Please update if statements in the code.');

 89% Do weighted average of signals based on (inverse) entropy to yield a
 90% signal (heartrate and breathing).
 91
 92% power = (a^2+b^2)/N
 93% Should be same as 1/N*abs(Y)^2
 94npts = size(Abody,1);
 95Fheartrate = zeros(floor(npts/2), size(Ahead,2));
 96Fbreathing = zeros(floor(npts/2), size(Abody,2));
 97

100freq = fps/npts * (0:floor(npts/2)-1);

102Y = fft(Ahead, npts);
103Pyy = Y.*conj(Y)/npts;
104freq = fps/npts * (0:floor(npts/2)-1);
105Fheartrate = Pyy(1:floor(npts/2),:);

107% [freq, Fbreathing] = plot_power_spectrum(Abody, 60);
108% [freq, Fheartrate] = plot_power_spectrum(Ahead_filt, 60);

112Fheartrate(1,:) = 0;

121RRentr = get_spectral_entropy(Fbreathing);

124% trust neighborhoods of high weight
125B = 1-reshape(wrr,size(Bbody(:,:,1)));
126% B2 = colfilt(B,[2 2],'sliding',@min);

131    B(i,j) = min(patch(:));
132  end
133end
134wrr = B(:);
135% !!
136wrr = ones(size(wrr))./numel(wrr);

139% Use fft to get rid of phase differences between signals.
