1function [hr,rr] = extract_vitals(frames, block_size)
  2
  3% assert(isa(frames,'double'))
  4% assert(isa(block_size,'double'))
  5
  6if nargin < 2
  7  block_size = 6;
  8end
  9
 10% tic
 11% frames = imrotate(frames,180);
 12% toc
 13
 14fprintf(1,'Finding face...')
 15% Need to make face pointing up so it works with Viola-Jones.
 16frame = flipud(frames(:,:,1));
 17% Viola-Jones cascade face detector. Min size chosen empirically based on
 18% optical setup.
 19faceDetector = vision.CascadeObjectDetector('MinSize',[75 75]);
 20% faceDetector = vision.CascadeObjectDetector('ProfileFace','MinSize',[75 75]);
 21face_rect = step(faceDetector, frame);
 22
 23if isempty(face_rect)
 24%   warning('No face detected. Using coarse ROIs.')
 25  body = frames(2:304,2:678,:);
 26  head = frames(443:851, 110:521, :);
 27else
 28  [H,W,T] = size(frames);
 29  x = face_rect(1);
 30  y = H-face_rect(2);
 31  w = face_rect(3);
 32  h = face_rect(4);
 33  fprintf(1,'Done\n')
 34
 35  % Draw the returned bounding box around the detected face.
 36%   videoOut = insertObjectAnnotation(frame,'rectangle',face_rect,'Face');
 37%   figure, imshow(videoOut), title('Detected face');
 38
 39  % Expand the head ROI so we get the neck as well.
 40%   head = frames(y-h*1.5 : y+h*.10, x:x+w, :);
 41  head = frames(y-h : y, x:x+w, :);
 42  % The rest is body.
 43  body = frames(1 : y-h*2, 1 : end, :);
 44end
 45  
 46Bbody = imresize(body,1/block_size,'box');
 47Rbody = reshape(permute(Bbody,[3 1 2]),size(Bbody,3),[]);
 48Abody = -log(1+Rbody);
 49
 50Bhead = imresize(head,1/block_size,'box');
 51Rhead = reshape(permute(Bhead,[3 1 2]),size(Bhead,3),[]);
 52Ahead = -log(1+Rhead);
 53
 54[T,~] = size(Ahead);
 55
 56%%
 57% breathing = mean(Abody,2);
 58% heartrate = mean(Ahead,2);
 59
 60% Get rid of breathing component of heartrate signal.
 61low_freq = 30/60;
 62high_freq = 350/60;
 63% x_ts = timeseries(Ahead, [0:T-1]./60);
 64% x_filt = idealfilter(x_ts, [low_freq high_freq], 'pass');
 65% Ahead_filt = x_filt.Data;
 66
 67fs = 60;
 68nyq = fs/2;
 69[b,a] = butter(3,[low_freq high_freq]./nyq,'bandpass');
 70Ahead_filt = zeros(size(Ahead));
 71for i = 1:size(Ahead_filt,2)
 72  Ahead_filt(:,i) = filter(b, a, Ahead(:,i));
 73end
 74
 75
 76
 77
 78% Do weighted average of signals based on (inverse) entropy to yield a
 79% signal (heartrate and breathing).
 80
 81% power = (a^2+b^2)/N
 82% Should be same as 1/N*abs(Y)^2
 83npts = size(Abody,2);
 84Fheartrate = zeros(floor(npts/2), size(Abody,2));
 85Fbreathing = zeros(floor(npts/2), size(Abody,2));
 86
 87Y = fft(Abody, npts);
 88Pyy = Y.*conj(Y)/npts;
 89freq = 60/npts * (0:floor(npts/2)-1);
 90Fbreathing = Pyy(1:floor(npts/2),:);
 91Y = fft(Ahead, npts);
 92Pyy = Y.*conj(Y)/npts;
 93freq = 60/npts * (0:floor(npts/2)-1);
 94Fheartrate = Pyy(1:floor(npts/2),:);
 95
 96% [freq, Fbreathing] = plot_power_spectrum(Abody, 60);
 97% [freq, Fheartrate] = plot_power_spectrum(Ahead_filt, 60);
 98for i = 1:size(Fbreathing,2)
 99  Fbreathing(1,i) = 0;
100  Fheartrate(1,i) = 0;
101end
102% Fbreathing(1,:) = zeros(1, size(Fbreathing,2));
103% Fheartrate(1,:) = zeros(1, size(Fheartrate,2));
104
105HRentr = get_spectral_entropy(Fheartrate);
106whr = 1-HRentr;
107whr = whr./sum(whr);
108heartrate = Ahead_filt * whr';
109
110RRentr = get_spectral_entropy(Fbreathing);
111wrr = 1-RRentr;
112wrr = wrr./sum(wrr);
113% Use fft to get rid of phase differences between signals.
114F = abs(fft(Abody));
115F(1,:) = 0;
116breathing = ifft(F * wrr');
117
118sigmam = 1.4e-3;
119sigmap = 2.3e-5;
120R = exp(-heartrate)-1;
121xhat = kalmanfilt(R', sigmap, sigmam);
122heartrate = -log(xhat+1);
123
124[~,maxidx] = max(whr);
125heartrate_best = Ahead_filt(:,maxidx);
126R = exp(-heartrate_best)-1;
127xhat = kalmanfilt(R', sigmap, sigmam);
128heartrate_best = -log(xhat+1);
129% 
130% figure;
131% subplot(1,2,1),
132% [H,W,~] = size(Bhead);
133% imshow(flipud((reshape(whr,[H W]))),[])
134% title('Pulse weights')
135% subplot(1,2,2)
136% [H,W,~] = size(Bbody);
137% imshow(flipud(reshape(wrr,[H W])),[]);
138% title('Breathing weights')
139
140%% Find vitals
141[freq, Fbreathing] = plot_power_spectrum(breathing, 60);
142[freq, Fheartrate] = plot_power_spectrum(heartrate, 60);
143
144Fbreathing(1,:) = 0;
145Fheartrate(1,:) = 0;
146
147[~,maxidx] = max(Fheartrate);
148hr = freq(maxidx);
149[~,maxidx] = max(Fbreathing);
150rr = freq(maxidx);
151
152% t = [0:numel(heartrate_best)-1] ./ 60;
153% figure;
154% subplot(2,2,1), plot(t,heartrate_best), title('Heart rate')
155% hold on, plot(t,heartrate,'--k')
156% xlabel('time (s)')
157% ylabel('blood volume (a.u.)')
158% legend('strongest signal','weighted average')
159% 
160% subplot(2,2,2), plot(t,breathing), title('Breathing')
161% xlabel('time (s)')
162% ylabel('breathing amplitude (a.u.)')
163% 
164% subplot(2,2,3), plot(60*freq,Fheartrate), title(sprintf('HR=%u bpm',round(hr*60)))
165% [~,maxidx] = max(Fheartrate);
166% hold on, plot(60*freq(maxidx),Fheartrate(maxidx),'or')
167% xlim([0 300])
168% xlabel('frequency (1/min)')
169% ylabel('spectral power (dB)')
170% 
171% subplot(2,2,4), plot(60*freq,Fbreathing), title(sprintf('RR=%u bpm',round(rr*60)))
172% [~,maxidx] = max(Fbreathing);
173% hold on, plot(60*freq(maxidx),Fbreathing(maxidx),'or')
